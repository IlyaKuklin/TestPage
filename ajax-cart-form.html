Здесь есть HTML-элемент <form>, но кнопка имеет type="button", поэтому событие submit не происходит. Товар добавляется через AJAX-запрос.
Код перехвата AJAX-запроса
<script>
  window.dataLayer = window.dataLayer || [];

  (function () {
    const originalFetch = window.fetch;

    window.fetch = function () {
      const fetchArguments = arguments;
      const requestUrl = String(fetchArguments[0]);

      return originalFetch.apply(this, fetchArguments).then(function (response) {
        const clonedResponse = response.clone();

        if (requestUrl.includes('cart-response.json') && response.ok) {
          clonedResponse.json().then(function (data) {
            if (data && data.success === true) {
              window.dataLayer.push({
                event: 'ajax_add_to_cart_success',
                product_id: 'course-web-analytics',
                product_name: 'Курс по веб-аналитике'
              });

              console.log('dataLayer push: ajax_add_to_cart_success');
            }
          });
        }

        return response;
      });
    };
  })();
</script>
